class Solution {
public:
    int removeDuplicates(vector<int>& nums) {
        int n=nums.size();
        int t=0;
        for(int i=1;i<n;i++){
            if(nums[i-1]==nums[i]){
                t++;
            }
            else{
                nums[i-t]=nums[i];
            }
        }
        return n-t;
    }
};
