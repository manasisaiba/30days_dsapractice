class Solution {
public:
    int fxn(int n){
        int temp=n;
        int sum=0;
        
        while(n!=0){
            int r=n%10;
            sum+=r*r;
            n=n/10;
        }
        return sum;
    }
    bool isHappy(int n) {
        while(n!=1 && n!=4){
            n=fxn(n);
        }
        return n==1;
    }
};
