vector<int> getPositiveCumulativeSum(vector<int> &arr) {
	// add your logic here
	int n=arr.size();
	
	vector<int>v(n);
	vector<int>ans;
	v[0]=arr[0];
	for(int i=1;i<arr.size();i++){
		v[i]=v[i-1]+arr[i];		
	}
	for(int i=0;i<v.size();i++){
		if(v[i]>0){
			ans.push_back(v[i]);
		}
	}
	return ans;
}
